		}
    }
}

class Solution {
    public void rotate(int[] nums, int k) {

        int redefineK = k % nums.length;

        reverse(nums, 0, nums.length - 1);
        reverse(nums, 0, redefineK - 1);
        reverse(nums, redefineK, nums.length - 1);
    }

    public void reverse(int[] list, int start, int end) {
        while (start <= end) {
            int temp = list[start];
            list[start++] = list[end];
            list[end--] = temp;
        }
    }
}
